stretchColumns = "1,2" 			//使某个控件填满这一行多余的空间，也是设置一行有多少列的方法，如果一行有四列就设置"0,1,2,3"

内部属性(子控件使用的属性)
colunm = "1"				//设置子控件在某列的位置

span = "2"				//设置子控件占几列
